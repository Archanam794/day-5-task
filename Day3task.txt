					JSON  ITERATEN FOR-EACH LOOP
code
var toyshop={
  "doll1":"teddy",
  "doll2":"barbie",
  "doll3":"spiderman",
  "doll4":"nippon",
  "vehicle":["car","bike","truck","firetruck","train","aeroplane"]
};
Object.keys(toyshop).forEach((key)=>{
    console.log(key,toyshop[key]);
})

Output:
doll1 teddy
doll2 barbie
doll3 spiderman
doll4 nippon
vehicle [ 'car', 'bike', 'truck', 'firetruck', 'train', 'aeroplane' ]
